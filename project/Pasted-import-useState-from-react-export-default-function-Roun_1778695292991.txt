import { useState } from "react";

export default function RoundTracker({ go, setUser }) {
  const [course, setCourse] = useState("");
  const [comp, setComp] = useState(false);
  const [score, setScore] = useState("");
  const [fh, setFh] = useState("");
  const [gir, setGir] = useState("");
  const [putts, setPutts] = useState("");
  const [scramble, setScramble] = useState("");

  const submit = () => {
    const round = {
      course,
      comp,
      score,
      fh,
      gir,
      putts,
      scramble,
    };

    setUser((prev) => ({
      ...prev,
      lastRound: round,
    }));

    go("dashboard");
  };

  return (
    <div className="min-h-screen flex items-center justify-center bg-[#ece8e1] p-6">
      <div className="bg-white p-10 rounded-3xl shadow-xl w-full max-w-xl">

        <h1 className="text-2xl mb-6">Submit Round</h1>

        <input
          placeholder="Course"
          className="w-full border p-3 mb-3 rounded-xl"
          value={course}
          onChange={(e) => setCourse(e.target.value)}
        />

        <label className="block mb-3">
          <input type="checkbox" onChange={(e) => setComp(e.target.checked)} />
          {" "}Competition round?
        </label>

        <input placeholder="Score" className="w-full border p-3 mb-3 rounded-xl"
          value={score} onChange={(e) => setScore(e.target.value)} />

        <input placeholder="Fairways Hit (0–14)" className="w-full border p-3 mb-3 rounded-xl"
          value={fh} onChange={(e) => setFh(e.target.value)} />

        <input placeholder="GIR (0–18)" className="w-full border p-3 mb-3 rounded-xl"
          value={gir} onChange={(e) => setGir(e.target.value)} />

        <input placeholder="Putts" className="w-full border p-3 mb-3 rounded-xl"
          value={putts} onChange={(e) => setPutts(e.target.value)} />

        <input placeholder="Scramble saves" className="w-full border p-3 mb-6 rounded-xl"
          value={scramble} onChange={(e) => setScramble(e.target.value)} />

        <button onClick={submit} className="w-full bg-black text-white py-3 rounded-xl">
          Submit Round
        </button>
      </div>
    </div>
  );
}
